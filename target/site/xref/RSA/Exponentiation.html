17      /**
18       * Returns the remainder when an integer x raised to the yth power, x^y, is divided by an integer n.
19       *
20       * @param x the base of the exponentiation
21       * @param y the exponent of the exponentiation
22       * @param n the modulus of the exponentiation
23       * @return the remainder of the exponentiation
24       */
25      public static BigInteger modPower(BigInteger x, BigInteger y, BigInteger n) {
26  
27          BigInteger res = BigInteger.ONE; // Initialize result
28  
29          //Update x if it is more than or
30          // equal to n
31          x = x.mod(n);
32  
33          while (y.compareTo(BigInteger.ZERO) > 0) {
34  
35              // If y is odd, multiply x with result
36              if (y.and(BigInteger.ONE).equals(BigInteger.ONE))
37                  res = res.multiply(x).mod(n);
38  
39              // y must be even now
40              y = y.shiftRight(1); // y = y/2
41              x = x.multiply(x).mod(n);
42          }
43  
44          return res;
45      }

47      /**
48       * Returns the greatest exponent of the parameter's binary representation.
49       * Used for the method {@link #getExponents(BigInteger)}
50       */
51      static int getGreatestExponent(BigInteger e) {
52          int i = 0;
53          while (e.compareTo(BigInteger.valueOf((long) Math.pow(2, i))) >= 0) {
54              i++;
55          }
56          return i;
57      }
58  
59      /**
60       * Calculates the powers of two of the given integer's representation.
61       *
62       * @param e the given integer
63       * @return the exponents of two
64       */
65      static List<Integer> getExponents(BigInteger e) {
66          List<Integer> exponents = new ArrayList<>();
67          int i = getGreatestExponent(e);
68  
69          exponents.add(i - 1 > 0 ? i - 1 : 0);
70          e = e.subtract(BigInteger.valueOf((long)Math.pow(2, i - 1 > 0 ? i - 1 : 0)));
71          int j = i-2;
72          while (j >= 0) {
73              if (e.compareTo(BigInteger.valueOf((long) Math.pow(2, j))) >= 0) {
74                  exponents.add(j);
75                  e = e.subtract(BigInteger.valueOf((long)Math.pow(2, j)));
76              }
77              j--;
78          }
79  
80          return exponents;
81      }

83      /**
84       * The remainders got from the given integer's division with remainder.
85       *
86       * @param greatestExponent greatest exponent got from {@link #getGreatestExponent(BigInteger)}
87       * @param g the given integer
88       * @param n the modulus used for the division with remainder
89       * @return the remainders used to calculate the product with {@link #getProduct(List, BigInteger[])}
90       */
91      static BigInteger[] getExponentRemainders(int greatestExponent, BigInteger g, BigInteger n) {
92          BigInteger[] remainders = new BigInteger[greatestExponent];
93          remainders[0] = g;
94          for (int i = 1; i < greatestExponent; i++) {
95              try {
96                  remainders[i] = (remainders[i-1].pow(2)).mod(n);
97              } catch (ArithmeticException e) {
98                  logger.error("Can't divide by 0.");
99                  break;
100             }
101         }
102 
103         return remainders;
104     }

106     /**
107      * Multiplies all the exponents.
108      *
109      * @param exponents the exponents, representing the remainders to be multiplied
110      * @param remainders the remainders to be multiplied
111      * @return the final result of the modular exponentiation.
112      */
113      static BigInteger getProduct(List<Integer> exponents, BigInteger[] remainders) {
114         BigInteger product = BigInteger.ONE;
115 
116         for (int i : exponents) {
117             product = product.multiply((remainders[i]));
118         }
119 
120         return product;
121     }

123     /**
124      * Driver of the modular exponentiation.
125      * Returns the remainder when an integer g raised to the eth power, g^e, is divided by an integer n.
126      *
127      * @param g the base integer
128      * @param e the exponent
129      * @param n the modulus
130      * @return the remainder of the exponentiation
131      */
132     public static BigInteger getRemainder(BigInteger g, BigInteger e, BigInteger n) {
133         if (g.equals(BigInteger.ZERO))
134             return BigInteger.ZERO;
135         else if (n.compareTo(BigInteger.ZERO) <= 0)
136             throw new ArithmeticException("Modulus should be positive.");
137         else {
138             try {
139                 BigInteger product = getProduct(getExponents(e), getExponentRemainders(getGreatestExponent(e), g, n));
140                 return product.mod(n);
141             } catch (ArithmeticException exc) {
142                 logger.error("Can't divide by 0.");
143             }
144             return BigInteger.valueOf(-1);
145         }
146     }
