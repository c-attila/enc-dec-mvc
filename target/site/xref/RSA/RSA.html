27      /** Converts the <code>m</code> string to an array of ASCII codes.*/
28      static BigInteger[] stringToArray(String m) {
29          BigInteger[] c = new BigInteger[m.length()];
30          for (int i = 0; i < m.length(); i++)
31              c[i] = BigInteger.valueOf(m.charAt(i));

33          return c;

36      /** Converts the <code>c</code> array of ASCII codes to string.*/
37      static String ASCIIToString(BigInteger[] c) {
38          String m = "";
39          for (BigInteger bigInteger : c) m = m + (char) bigInteger.intValue();
40          return m;

50      /**
51       * Encodes the <code>m</code> message using the <code>c^e mod n</code> formula.
52       * It represents the message as ASCII codes.
53       *
54       * @param m the message
55       * @param e the exponent
56       * @param n the modulus
57       * @return the encoded message as ASCII codes
58       */
59      static BigInteger[] encodeMessage(String m, BigInteger e, BigInteger n) {
60          BigInteger[] c = stringToArray(m);
61          for (int i = 0; i < c.length; i++)
62              c[i] = Exponentiation.modPower(c[i], e, n);
63  
64          return c;
65      }

67      /** Returns a random big prime. */
68      public static BigInteger pickPrime() {
69  
70          /** the bit-length of the prime number */
71          int bitLength = 512;
72          BigInteger rand = new BigInteger(bitLength, new Random());
73          rand = rand.setBit(0);
74          rand = rand.setBit(bitLength - 1);
75          do
76              rand = rand.add(BigInteger.TWO);
77          while (!primalityTest.isPrime(rand, 10));
78  
79          return rand;
80      }

82      /**
83       * Method implementing the Chinese remainder theorem for decoding.
84       *
85       * @param c the encoded message, represented with ASCII codes
86       * @param d the private key to decode the message
87       * @param p random big prime from the EncDecMVC.RSA algorithm
88       * @param q random big prime from the EncDecMVC.RSA algorithm
89       * @return the message decoded, represented with ASCII codes
90       */
91      static BigInteger[] decodeEncrypted(BigInteger[] c, BigInteger d, BigInteger p, BigInteger q) {
92          BigInteger cp, cq;
93          BigInteger[][] table = GCD.getTable(p, q);
94          for (int i = 0; i < c.length; i++) {
95              cp = Exponentiation.modPower(c[i], d.mod(p.subtract(BigInteger.ONE)), p);
96              cq = Exponentiation.modPower(c[i], d.mod(q.subtract(BigInteger.ONE)), q);
97              c[i] = (p.multiply(table[2][table[2].length-1]).multiply(cp)).add(q.multiply(table[3][table[3].length-1]).multiply(cq)).mod(BigInteger.valueOf(1457));
98          }
99  
100         return c;
101     }

106     public static String getRandomKeys() {
107         BigInteger p = pickPrime();
108         BigInteger q = pickPrime();
109 
110         BigInteger n = p.multiply(q);
111         BigInteger fin = (p.subtract(BigInteger.ONE)).multiply(q.subtract(BigInteger.ONE));
112         BigInteger e;
113         BigInteger[][] EEtable;
114         do {
115             e = new BigInteger(fin.bitLength(), new Random());
116             EEtable = GCD.getTable(fin, e);
117         } while ((e.compareTo(fin) >= 0 || e.compareTo(BigInteger.ONE) <= 0) || !(EEtable[0][EEtable[0].length - 1]).equals(BigInteger.ONE));
118 
119         //BigInteger d = e.modInverse(fin);
120         BigInteger d = EEtable[3][EEtable[3].length-1];
121         while (d.compareTo(BigInteger.ZERO) < 0)
122             d = d.add(fin);
123 
124         String keysAsText = "Private key (d):        " + d + "\n" + "Public exponent (e):    " + e + "\n" + "Modulus (n):            " + n + "\n";
125 
126         return keysAsText;

129     /**
130      * Implements the EncDecMVC.RSA algorithm to generate random keys and encodes/decodes with it.
131      * Uses the Chinese remainder theorem from the
132      * {@link #decodeEncrypted(BigInteger[], BigInteger, BigInteger, BigInteger)} method
133      * and the Extended Euclidean Algorithm from the <code>GCD</code> class.
134      *
135      * @param m the message to be encoded/decoded
136      * @return the <code>m</code> string encoded and decoded back
137      * */
138     public static String encodeDecode(String m){
139         BigInteger p = pickPrime();
140         BigInteger q = pickPrime();
141 
142         BigInteger n = p.multiply(q);
143         BigInteger fin = (p.subtract(BigInteger.ONE)).multiply(q.subtract(BigInteger.ONE));
144         BigInteger e;
145         BigInteger[][] EEtable;
146         do {
147             e = new BigInteger(fin.bitLength(), new Random());
148             EEtable = GCD.getTable(fin, e);
149         } while ((e.compareTo(fin) >= 0 || e.compareTo(BigInteger.ONE) <= 0) || !(EEtable[0][EEtable[0].length - 1]).equals(BigInteger.ONE));
150 
151         //BigInteger d = e.modInverse(fin);
152         BigInteger d = EEtable[3][EEtable[3].length-1];
153         while (d.compareTo(BigInteger.ZERO) < 0)
154             d = d.add(fin);
155 
156         System.out.println("The message:    " + m);
157         System.out.print("The message in ASCII:    ");
158         printAscii(m);
159         BigInteger[] c = encodeMessage(m, e, n);
160         System.out.print("The message encoded:    ");
161         printC(c);
162         c = decodeEncrypted(c, d, p, q);
163         System.out.print("The message decoded:    ");
164         printC(c);
165         System.out.print("The message:    ");
166         printChar(c);
167 
168         return ASCIIToString(c);
169     }
