6   /** Primality test, implementing the Miller-Rabin algorithm. */
7   public class primalityTest {
8   
9       /**
10       * This function is called for all k trials. It returns false if n is composite
11       * and returns false if n is probably prime.
12       *
13       * @param d an odd number such that d*2^r = n-1 for some r >= 1
14       * @param n the number to be tested
15       * @return true if <code>n</code> is probably prime, false if <code>n</code> is composite
16       */
17      static boolean millerTest(BigInteger d, BigInteger n) {
18  
19          // Pick a random number in [2..n-2]
20          // Corner cases make sure that n > 4
21          BigInteger a;
22          do {
23              a = new BigInteger(n.subtract(BigInteger.ONE).bitLength(), new Random());
24          } while ((n.subtract(BigInteger.ONE).compareTo(BigInteger.valueOf(2)) > 0 ? a.compareTo(n.subtract(BigInteger.ONE)) >= 0 : a.compareTo(BigInteger.valueOf(3)) >= 0) || a.compareTo(BigInteger.valueOf(2)) < 0);
25  
26  
27          // Compute a^d % n
28          BigInteger x = Exponentiation.modPower(a, d, n);
29  
30          if (x.equals(BigInteger.ONE) || x.equals(n.subtract(BigInteger.ONE)))
31              return true;
32  
33          // Keep squaring x while one of the
34          // following doesn't happen
35          // (i) d does not reach n-1
36          // (ii) (x^2) % n is not 1
37          // (iii) (x^2) % n is not n-1
38          while (!d.equals(n.subtract(BigInteger.ONE))) {
39              x = Exponentiation.modPower(x, BigInteger.TWO, n);
40              d = d.multiply(BigInteger.TWO);
41  
42              if (x.equals(BigInteger.ONE))
43                  return false;
44              if (x.equals(n.subtract(BigInteger.ONE)))
45                  return true;
46          }
47  
48          // Return composite
49          return false;
50      }
51  
52      // It returns false if n is composite
53      // and returns true if n is probably
54      // prime. k is an input parameter that
55      // determines accuracy level. Higher
56      // value of k indicates more accuracy.
57  
58      /**
59       * Probes the probability of the given number being prime.
60       *
61       * @param n the given number to prove prime
62       * @param k determines the accuracy level
63       * @return false if <code>n</code> is composite, true if <code>n</code> is probably prime
64       */
65      public static boolean isPrime(BigInteger n, int k) {
66  
67          // Corner cases
68          if (n.compareTo(BigInteger.ONE) <= 0 || n.equals(BigInteger.valueOf(4)))
69              return false;
70          if (n.compareTo(BigInteger.valueOf(3)) <= 0)
71              return true;
72  
73          // Find r such that n = 2^d * r + 1
74          // for some r >= 1
75          BigInteger d = n.subtract(BigInteger.ONE);
76  
77          while (Exponentiation.modPower(d, BigInteger.ONE, BigInteger.TWO).equals(BigInteger.ZERO))
78              d = d.divide(BigInteger.TWO);
79  
80          // Iterate given nber of 'k' times
81          for (int i = 0; i < k; i++)
82              if (!millerTest(d, n))
83                  return false;
84  
85          return true;
86      }
