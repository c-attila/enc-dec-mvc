8       /** Prints the 2D array <code>table</code>. */
9       static void printTable(BigInteger[][] table) {
10          for (int i = 0; i < 4; i++) {
11              for (int j = 0; j < table[i].length; j++)
12                  System.out.print(table[i][j] + " ");
13              System.out.println();
14          }
15      }

17      /**
18       * Calculates the length of the <code>table</code> used for the EEA.
19       *
20       * @param r first parameter of the EEA
21       * @param e second parameter of the EEA
22       * @return the length for the EEA table
23       */
24      static int getLength(BigInteger r, BigInteger e) {
25          int n = 0;
26          BigInteger etemp;
27          while (e.compareTo(BigInteger.ZERO) > 0) {
28              etemp = e;
29              e = r.mod(e);
30              r = etemp;
31              n++;
32          }
33          return n;
34      }
35  
36      /**
37       * Produces the <code>table</code> got from the EEA.
38       * <code>table[2][length]</code> and <code>table[3][length]</code> is
39       * already multiplied with -1^n and -1^(n+1).
40       *
41       * @param r first parameter of the EEA
42       * @param e second parameter of the EEA
43       * @return the <code>table</code>, produced by the EEA
44       */
45      public static BigInteger[][] getTable(BigInteger r, BigInteger e) {
46          if (r.equals(BigInteger.ZERO) && e.equals(BigInteger.ZERO))
47              throw new ArithmeticException("GCD of 0 and 0 is not defined.");
48          else if (r.equals(BigInteger.ZERO)) {
49              BigInteger[][] res = new BigInteger[1][1];
50              res[0][0] = e;
51              return res;
52          }
53          else if (e.equals(BigInteger.ZERO)) {
54              BigInteger[][] res = new BigInteger[1][1];
55              res[0][0] = r;
56              return res;
57          }
58          int n = getLength(r, e);
59          BigInteger[][] table = new BigInteger[4][n + 1];
60          table[0][0] = r;
61          table[0][1] = e;
62          table[1][1] = r.divide(e);
63          table[2][0] = BigInteger.ONE;
64          table[2][1] = BigInteger.ZERO;
65          table[3][0] = BigInteger.ZERO;
66          table[3][1] = BigInteger.ONE;
67  
68          for (int i = 2; i < n + 1; i++) {
69              table[0][i] = table[0][i - 2].mod(table[0][i - 1]);
70              table[1][i] = table[0][i - 1].divide(table[0][i]);
71              table[2][i] = (table[1][i-1].multiply(table[2][i - 1])).add(table[2][i - 2]);
72              table[3][i] = (table[1][i-1].multiply(table[3][i - 1])).add(table[3][i - 2]);
73          }
74          table[2][n] = BigInteger.valueOf(-1).pow(n).multiply(table[2][n]);
75          table[3][n] = BigInteger.valueOf(-1).pow(n+1).multiply(table[3][n]);
76  
77          return table;
78      }
